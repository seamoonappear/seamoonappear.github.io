<script>
// 简单直接的导航功能实现
(function() {
  // 等待页面完全加载
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initMediaTabs);
  } else {
    initMediaTabs();
  }
  
  function initMediaTabs() {
    console.log('初始化媒体标签页');
    
    // 获取所有导航按钮
    const navButtons = document.querySelectorAll('.nav-btn');
    const tabPanes = document.querySelectorAll('.tab-pane');
    
    // 为每个按钮添加点击事件
    navButtons.forEach(button => {
      // 移除可能存在的旧事件监听器
      button.replaceWith(button.cloneNode(true));
    });
    
    // 重新获取按钮引用
    const newButtons = document.querySelectorAll('.nav-btn');
    
    newButtons.forEach(button => {
      button.addEventListener('click', function(e) {
        e.preventDefault();
        e.stopPropagation();
        
        const targetTab = this.getAttribute('data-tab');
        console.log('点击标签:', targetTab);
        
        // 移除所有按钮的active类
        newButtons.forEach(btn => {
          btn.classList.remove('active');
        });
        
        // 移除所有内容面板的active类
        tabPanes.forEach(pane => {
          pane.classList.remove('active');
        });
        
        // 添加active类到当前按钮和对应内容
        this.classList.add('active');
        const targetPane = document.getElementById(targetTab);
        if (targetPane) {
          targetPane.classList.add('active');
          console.log('成功切换到标签:', targetTab);
        } else {
          console.error('找不到目标标签页:', targetTab);
        }
      });
    });
    
    // 检查是否有活动的标签页，如果没有则激活第一个
    const activePanes = document.querySelectorAll('.tab-pane.active');
    if (activePanes.length === 0 && tabPanes.length > 0) {
      tabPanes[0].classList.add('active');
    }
    
    console.log('媒体标签页初始化完成');
  }
})();
</script>
